PowerShell Enumeration Scripts

PowerShell Enumeration Automation Post-Exploitation

Overview

Custom PowerShell scripts generated and modified for system enumeration and shell access testing while Windows Defender remained active. Includes system info gathering, network scanning, and credential hunting.

System Information Script

function Get-SystemInfo {
  $info = @{}
  $info.Hostname = $env:COMPUTERNAME
  $info.Username = $env:USERNAME
  $info.OS = (Get-WmiObject Win32_OperatingSystem).Caption
  $info.Arch = (Get-WmiObject Win32_ComputerSystem).SystemType
  $info.IP = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object {$_.IPAddress -notmatch "127.*"})[0].IPAddress
  return $info
}

Credential Hunting Script

function Find-Credentials {
  $paths = @(
    "$env:USERPROFILE\.aws\credentials",
    "$env:USERPROFILE\.ssh\*",
    "$env:APPDATA\Microsoft\Windows\PowerShell\PSReadLine\ConsoleHost_history.txt",
    "C:\Windows\System32\config\*"
  )
  foreach ($path in $paths) {
    if (Test-Path $path) {
      Write-Host "[+] Found: $path" -ForegroundColor Green
    }
  }
}

Network Scanner Script

function Invoke-PortScan {
  param($Subnet = "192.168.1", $Ports = @(22,80,443,445,3389,5985))
  1..254 | ForEach-Object {
    $ip = "$Subnet.$_"
    foreach ($port in $Ports) {
      $result = Test-NetConnection -ComputerName $ip -Port $port -WarningAction SilentlyContinue
      if ($result.TcpTestSucceeded) {
        Write-Host "[+] $ip : $port is open" -ForegroundColor Green
      }
    }
  }
}
